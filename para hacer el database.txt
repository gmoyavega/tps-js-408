auth.onAuthStateChanged((user) => {
  if (user) {
    // El usuario ha iniciado sesión
    console.log("Usuario actual:", user.email);
    // Puedes usar user.uid para acceder a datos protegidos en la base de datos
    // ej: database.ref(`usuarios/${user.uid}/perfil`).once('value', ...);
    document.getElementById('loginArea').style.display = 'none';
    document.getElementById('userDashboard').style.display = 'block';
  } else {
    // El usuario ha cerrado sesión o no ha iniciado
    console.log("No hay usuario activo.");
    document.getElementById('loginArea').style.display = 'block';
    document.getElementById('userDashboard').style.display = 'none';
  }
});

        // *** CONFIGURACIÓN DE FIREBASE ***
        // ¡DEBES REEMPLAZAR ESTOS VALORES CON LOS DE TU PROYECTO DE FIREBASE!
        const firebaseConfig = {
            apiKey: "TU_API_KEY_AQUI", 
            authDomain: "TU_DOMINIO.firebaseapp.com",
            projectId: "tu-proyecto-id",
            // ... otros campos si son necesarios (storageBucket, messagingSenderId, appId)
        };

        // Inicializar Firebase
        const app = firebase.initializeApp(firebaseConfig);
        const auth = firebase.auth();
        // ---------------------------------


        // Referencias a elementos del DOM
        const loginForm = document.getElementById('loginForm');
        const emailInput = document.getElementById('emailInput');
        const passwordInput = document.getElementById('passwordInput');
        const errorMessage = document.getElementById('error-message');

        loginForm.addEventListener('submit', (e) => {
            e.preventDefault(); // Evita el envío tradicional del formulario

            const email = emailInput.value;
            const password = passwordInput.value;
            errorMessage.textContent = ''; // Limpiar mensajes de error previos

            // Lógica de inicio de sesión de Firebase
            auth.signInWithEmailAndPassword(email, password)
                .then((userCredential) => {
                    // Inicio de sesión exitoso
                    console.log("Inicio de sesión exitoso. UID:", userCredential.user.uid);
                    
                    // *** ACCIÓN POST-LOGIN ***
                    // Redirigir al usuario a la página principal de la aplicación (Dashboard)
                    window.location.href = 'index.html'; // Cambia esto por la ruta de tu página principal
                })
                .catch((error) => {
                    // Manejo de errores
                    console.error("Error de login:", error);
                    let message = "Error desconocido.";

                    switch (error.code) {
                        case 'auth/user-not-found':
                            message = 'Usuario no encontrado. Verifica el correo.';
                            break;
                        case 'auth/wrong-password':
                            message = 'Contraseña incorrecta.';
                            break;
                        case 'auth/invalid-email':
                            message = 'El formato del correo es inválido.';
                            break;
                        default:
                            message = 'Error al iniciar sesión. Intenta nuevamente.';
                            break;
                    }
                    errorMessage.textContent = message;
                });
        });
    </script>

Instrucciones :
Reemplaza la Configuración:

 Tienes que ir a la Consola de Firebase de tu proyecto, obtener tu configuración y reemplazar los valores en la variable firebaseConfig dentro de la etiqueta <script>.

Habilita el Proveedor:

Asegúrate de haber habilitado "Correo electrónico/Contraseña" en la sección Authentication > Sign-in method de tu consola de Firebase.

Crea index.html:

Este código asume que tienes una página principal llamada index.html a la que se redirigirá al usuario después de iniciar sesión correctamente.

Crea registro.html (Opcional):

Si también quieres una página de registro separada, crea un archivo para manejar la función createUserWithEmailAndPassword().
